> Example request:

```javascript

    "http://127.0.0.1:8000/api/blacklists"
);

let params = {
    "nom": "nulla",
    "prenom": "commodi",
    "mail": "impedit",
    "id_univ": "qui",
};
Object.keys(params)
    .forEach(key => url.searchParams.append(key, params[key]));

let headers = {
    "Authorization": "Bearer {YOUR_AUTH_KEY}",
    "Content-Type": "application/json",
    "Accept": "application/json",
};


fetch(url, {
    method: "POST",
    headers,
}).then(response => response.json());
```
